    \IfFontExistsTF{Consolas}{\setmonofont{Consolas}}{\setmonofont{DejaVu Sans Mono}}
    %
    \IfFontExistsTF{Segoe UI}{
                              \newfontfamily{\fallbackfontSEG}{Segoe UI   }
    }{
                              \newfontfamily{\fallbackfontSEG}{DejaVu Sans}
    }
    %
    \IfFontExistsTF{Segoe UI Symbol}{
                              \newfontfamily{\fallbackfontSYM}{Segoe UI Symbol }
    }{
                              \newfontfamily{\fallbackfontSYM}{DejaVu Sans     }
    }
    %
    \IfFontExistsTF{Segoe UI Emoji}{
                              \newfontfamily{\fallbackfontEMOJI}{Segoe UI Emoji}
    }{
                              \newfontfamily{\fallbackfontEMOJI}{DejaVu Sans   }
    }
    \newfontfamily{\fallbackfontDEJA}{DejaVu Sans}
    %
    \DeclareTextFontCommand{\FallBackSEG  }{\fallbackfontSEG  }
    \DeclareTextFontCommand{\FallBackSYM  }{\fallbackfontSYM  }
    \DeclareTextFontCommand{\FallBackEMOJI}{\fallbackfontEMOJI}
    \DeclareTextFontCommand{\FallBackDEJA }{\fallbackfontDEJA }
    %
    \newunicodechar{➤}{\FallBackSYM{➤}}
